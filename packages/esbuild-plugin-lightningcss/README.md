## Install

```sh
npm i @chialab/esbuild-plugin-lightningcss -D
```

```sh
yarn add @chialab/esbuild-plugin-lightningcss -D
```

## Usage

```js
import esbuild from 'esbuild';
import lightningcssPlugin from '@chialab/esbuild-plugin-lightningcss';

await esbuild.build({
    plugins: [
        lightningcssPlugin(),
    ],
});
```
